! Роутер 1 (VPN Gateway 1)
!
! Шаг 1: Настройка IKEv1 политики (Phase 1)
crypto ikev1 policy 10
 encryption aes
 hash sha
 authentication pre-share
 group 2
 lifetime 86400

! Шаг 2: Настройка IPSec трансформ-сета (Phase 2)
crypto ipsec transform-set MY_TRANSFORM esp-aes esp-sha-hmac

! Шаг 3: Настройка Pre-shared ключа для аутентификации
crypto ikev1 key MY_SECRET_KEY address 203.0.113.1

! Шаг 4: Создание ACL для шифрования трафика
access-list 100 permit ip 192.168.1.0 0.0.0.255 10.0.0.0 0.0.0.255

! Шаг 5: Настройка криптокарты для туннеля
crypto map VPN-MAP 10 ipsec-isakmp
 set peer 203.0.113.1
 set transform-set MY_TRANSFORM
 match address 100

! Шаг 6: Применение криптокарты на внешний интерфейс
interface Ethernet0
 crypto map VPN-MAP

! Шаг 7: Настройка маршрута для VPN трафика
ip route 10.0.0.0 255.255.255.0 203.0.113.1
